페이지가 로딩된 지 얼마나 시간이 지났는 지 알려주고, 새로고침을 누르면 시간이 초기화된다

<!DOCTYPE html> <html lang="en"> <head> <style> </style> </head> <body> <button onclick="reroadPage()">새로고침</button><br> <script> let bodyDOM = document.getElementsByTagName('body')[0]; let newDiv = document.createElement('div'); let count = 0; bodyDOM.appendChild(newDiv); newDiv.innerHTML = count + '초가 지났습니다.'; //페이지가 열린 후 지난 초를 표시 setInterval(() => { count++; newDiv.innerHTML = count + '초가 지났습니다.'; }, 1000); //페이지를 새로고침한다 function reroadPage(){ location.reload(); } </script> </body> </html>

현재 시간을 표시하고, 오전이면 배경이 초록색, 오후면 배경이 오렌지색이 된다.

<!DOCTYPE html> <html lang="en"> <head> <style> </style> </head> <body> <script> let bodyDOM = document.getElementsByTagName('body')[0]; let newP = document.createElement('p'); let now = new Date(); //오전이면 배경이 초록색, 아니면 오렌지색 if(now.getHours < 12){ bodyDOM.style.backgroundColor = 'green'; }else{ bodyDOM.style.backgroundColor = 'orange'; } newP.innerText = now; bodyDOM.appendChild(newP); //1초마다 현재 시간을 표시 setInterval(() => { now = new Date(); newP.innerText = now; bodyDOM.appendChild(newP); if(now.getHours < 12){ bodyDOM.style.backgroundColor = 'green'; }else{ bodyDOM.style.backgroundColor = 'orange'; } }, 1000); </script> </body> </html>
