package kr.bit.controller; import org.springframework.stereotype.Controller; import org.springframework.web.bind.annotation.GetMapping; import org.springframework.web.bind.annotation.PostMapping; import org.springframework.web.bind.annotation.RequestMapping; import org.springframework.web.bind.annotation.RequestMethod; @Controller public class SpringController { // spring1의 get방식 @RequestMapping(value="/spring1", method=RequestMethod.GET) public String spring1() { return "spring1"; } // spring2의 post방식 @RequestMapping(value="/spring2", method=RequestMethod.POST) public String spring2() { return "spring2"; } // spring5의 get방식 @RequestMapping(value="/spring3", method=RequestMethod.GET) public String spring3Get() { return "spring3"; } // spring5의 post방식 @RequestMapping(value="/spring3", method=RequestMethod.POST) public String spring3Post() { return "spring3"; } // spring4의 get방식 @GetMapping("/spring4") public String spring4() { return "spring4"; } // spring5의 post방식 @PostMapping("/spring5") public String spring5() { return "spring5"; } // spring7의 get/post방식 @RequestMapping(value="/spring7", method= {RequestMethod.GET, RequestMethod.POST}) public String spring7() { return "spring7"; } // 해당 방식은 get방식에 우선권이 있기 때문에 get방식으로만 작동함 @GetMapping("/spring8") @PostMapping("/spring8") public String spring8() { return "spring8"; } }
